        let boardData = [];

        let completedBingos = 0; 


        function initGame() {

            const board = document.getElementById('bingo-board');

            board.innerHTML = '';

            boardData = [];

            completedBingos = 0;

            updateCounter();

            

            const shuffled = [...memberNames].sort(() => 0.5 - Math.random());

            const selected = shuffled.slice(0, 24);

            

            let nameIndex = 0;

            for (let i = 0; i < 25; i++) {

                const cell = document.createElement('div');

                cell.className = 'bingo-cell';

                

                if (i === 12) {

                    cell.textContent = 'FREE';

                    cell.classList.add('free', 'marked');

                    boardData.push({ id: i, name: 'FREE', marked: true });

                } else {

                    const name = selected[nameIndex++];

                    cell.textContent = name;

                    cell.addEventListener('click', () => toggleMark(i, cell));

                    boardData.push({ id: i, name: name, marked: false });

                }

                

                board.appendChild(cell);

            }

        }


        function toggleMark(index, element) {

            boardData[index].marked = !boardData[index].marked;

            if (boardData[index].marked) {

                element.classList.add('marked');

            } else {

                element.classList.remove('marked');

            }

            checkBingo();

        }


        function updateCounter() {

            document.getElementById('bingo-counter').textContent = `Bingo Count: ${completedBingos}`;

        }


        function checkBingo() {

            const size = 5;

            let currentBingoCount = 0;


            // 横

            for (let i = 0; i < size; i++) {

                if (boardData.slice(i * size, i * size + size).every(d => d.marked)) currentBingoCount++;

            }


            // 縦

            for (let i = 0; i < size; i++) {

                let col = [];

                for (let j = 0; j < size; j++) col.push(boardData[i + j * size]);

                if (col.every(d => d.marked)) currentBingoCount++;

            }


            // 斜め

            let diag1 = [];

            let diag2 = [];

            for (let i = 0; i < size; i++) {

                diag1.push(boardData[i * (size + 1)]);

                diag2.push(boardData[(i + 1) * (size - 1)]);

            }

            if (diag1.every(d => d.marked)) currentBingoCount++;

            if (diag2.every(d => d.marked)) currentBingoCount++;


            if (currentBingoCount > completedBingos) {

                completedBingos = currentBingoCount;

                updateCounter();

                showOverlay();

            } else if (currentBingoCount < completedBingos) {

                completedBingos = currentBingoCount;

                updateCounter();

            }

        }


        function showOverlay() {

            document.getElementById('bingo-overlay').classList.remove('hidden');

        }


        function closeOverlay() {

            document.getElementById('bingo-overlay').classList.add('hidden');

        }


        // 画像として保存する処理

        async function saveAsImage() {

            const target = document.getElementById('capture-target');

            const saveBtn = document.getElementById('save-btn');

            

            // ボタンを一時的に無効化

            saveBtn.disabled = true;

            saveBtn.innerText = "保存中...";


            try {

                const canvas = await html2canvas(target, {

                    backgroundColor: '#000',

                    scale: 2, // 高解像度で保存

                    logging: false,

                    useCORS: true

                });

                

                const link = document.createElement('a');

                link.download = `AnagumaBingo_${new Date().getTime()}.png`;

                link.href = canvas.toDataURL('image/png');

                link.click();

            } catch (err) {

                console.error("保存に失敗しました:", err);

            } finally {

                saveBtn.disabled = false;

                saveBtn.innerHTML = `

                    <svg xmlns="http://www.w3.org/2000/svg" width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4"/><polyline points="7 10 12 15 17 10"/><line x1="12" y1="15" x2="12" y2="3"/></svg>

                    画像として保存

                `;

            }

        }
